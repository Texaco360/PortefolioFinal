Meter met procent

<div class="meter">
  <div class="meter__body">
    <div class="meter__fill"></div>
    <div class="meter__cover"></div>
  </div>
</div>

.meter {
  width: 100%;
  max-width: 250px;
  font-family: "Roboto", sans-serif;
  font-size: 32px;
  color: #004033;
}

.meter__body {
  width: 100%;
  height: 0;
  padding-bottom: 50%;
  background: #6c757d;
  position: relative;
  border-top-left-radius: 100% 200%;
  border-top-right-radius: 100% 200%;
  overflow: hidden;
}

.meter__fill {
  position: absolute;
  top: 100%;
  left: 0;
  width: inherit;
  height: 100%;
  background: rgb(2,0,36);
  background: radial-gradient(circle, rgba(2,0,36,1) 0%, rgba(9,9,121,1) 35%, rgba(0,212,255,1) 100%);
  transform-origin: center top;
  transition: transform 0.2s ease-out;
}

.meter__cover {
  width: 75%;
  height: 150%;
  background: #ffffff;
  border-radius: 50%;
  position: absolute;
  top: 25%;
  left: 50%;
  transform: translateX(-50%);

  /* Tekst */
  display: flex;
  align-items: center;
  justify-content: center;
  padding-bottom: 25%;
  box-sizing: border-box;
}

const meterElement = document.querySelector(".meter");

function setMeterValue(meter, value) {
    if (value < 0 || value > 1) {
        return;
    }

    meter.querySelector(".meter__fill").style.transform = `rotate(${
        value / 2
    }turn)`;
    meter.querySelector(".meter__cover").textContent = `${Math.round(
        value * 100
    )}%`;
}

/** procent berekenen **/
setMeterValue(meterElement, 0.3);
/** 0.3 = 30% **/
/** 0.75 = 75% **/
